## 🔧 技术实现

### 短信服务实现
```javascript
// netlify/functions/send-sms.js
const sms = require('qcloudsms_js');

exports.handler = async (event) => {
  const { phone, action } = JSON.parse(event.body);
  
  if (action === 'send') {
    // 生成验证码
    const code = Math.random().toString().slice(2, 8);
    
    // 存储验证码到 Redis/数据库
    await storeVerificationCode(phone, code);
    
    // 发送短信
    await sendSMS(phone, code);
    
    return { statusCode: 200, body: JSON.stringify({ success: true }) };
  }
  
  if (action === 'verify') {
    // 验证验证码
    const isValid = await verifyCode(phone, event.code);
    return { statusCode: 200, body: JSON.stringify({ valid: isValid }) };
  }
};
```
